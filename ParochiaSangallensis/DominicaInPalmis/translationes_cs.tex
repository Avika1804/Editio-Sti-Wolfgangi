%%%% Preklady jednotlivych zpevu (nektere se opakuji, a je dobre mit je
% vsechny na jedne hromade)

\newcommand{\trHosanna}{\translatioCantus{Hosana Synu Davidovu! Požehnaný, jenž přichází ve jménu Páně. Král Izraele! Hosana na výsostech.}}

\newcommand{\trPueriPortantes}{\translatioCantus{Židovské děti nesly olivové ratolesti a~vyšly, aby se setkali s Pánem; volaly a~říkaly: Hosana na výsostech.}}

\newcommand{\trPueriVestimenta}{\translatioCantus{Židovské děti položily na cestu své pláště a~volaly: Hosana Synu Davidovu! Požehnaný, jenž přichází ve jménu Páně.}}

\newcommand{\trGloriaLaus}{\translatioCantus{Chvála a~sláva bud Tobě, Kriste, zachránce, králi, jemuž z dětských úst zbožné Hosanna zní!}}

\newcommand{\trGloriaLausI}{\translatioCantus{Králem jsi Izraele, Davida potomkem slavným, ve jménu Páně jenž nám požehnat přicházíš sám.}}

\newcommand{\trGloriaLausII}{\translatioCantus{Bez ustání tě na výši nebeské zástupy chválí, člověk smrtelný též s veškerým tvorstvem Tvým.}}

\newcommand{\trGloriaLausIII}{\translatioCantus{Národ hebrejský v ústrety s palmami vycházel Tobě, s chválou a~sliby však my, ejhle, jdeme ti vstříc.}}

\newcommand{\trGloriaLausIV}{\translatioCantus{Před tvým utrpením oni ti vzdávali chválu, nám však jsi vládce a~král, jemuž náš chvalozpěv zní.}}

\newcommand{\trGloriaLausV}{\translatioCantus{V nich jsi měl zalíbení, nuž přijmi i modlitby tyto, králi dobrý ty náš, dobro jenž rád máš a~ctíš!}}

\newcommand{\trIngrediente}{\translatioCantus{Když Pán vstoupil do svatého města, židovské děti provolávaly vzkríšení života, s palmovými ratolestmi volaly: ,,Hosana na výsostech!\mbox{}`` \Vbardot{} Když lidé uslyšeli, že Ježíš přichází do Jeruzaléma, vyšli, aby ho potkali.}}

\newcommand{\trChristus}{\translatioCantus{Kristus \grestar{} se pro nás stal poslušný až k smrti, a~to k smrti na kříži! \Vbardot{} Proto také ho Bůh povýšil, a~dal mu jméno, které je nade všechno jméno.}}

\newcommand{\trCredo}{\translatioCantus{Věřím v~jednoho Boha, Otce
všemohoucího, Stvořitele nebe i~země, všeho viditelného i~neviditelného. 
Věřím v~jednoho Pána, Ježíše Krista, jednorozeného Syna Božího, který se
zrodil z~Otce přede všemi věky: Bůh z~Boha, Světlo ze Světla,
pravý Bůh z~pravého Boha, zrozený, nestvořený, jedné podstaty s Otcem: skrze něho
všechno je stvořeno. On pro nás lidi a~pro naši spásu sestoupil z~nebe.
Skrze Ducha svatého přijal tělo z~Marie Panny a~stal se člověkem. 
Byl za nás ukřižován, za dnů Poncia Piláta byl umučen a~pohřben. Třetího dne
vstal z~mrtvých podle Písma. Vstoupil do nebe, sedí po pravici Otce. A znovu
přijde, ve slávě, soudit živé i~mrtvé a~jeho království bude bez~konce. 
Věřím v~Ducha svatého, pána a~dárce života, který z~Otce i~Syna vychází,
s Otcem i~Synem je zároveň uctíván a~oslavován a~mluvil ústy proroků. 
Věřím v~jednu, svatou, všeobecnou, apoštolskou církev. Vyznávám jeden křest
na odpuštění hříchů. Očekávám vzkříšení mrtvých a~život budoucího věku.}}

\newcommand{\trIudica}{\translatioCantus{Obhajuj svou při, ochraň mě, neboť jsi mocný, Pane.}}

\newcommand{\trOratioDominica}{\translatioCantus{Otče náš, jenž jsi na
nebesích, posvěť se jméno tvé. Přijď království tvé. Buď vůle tvá jako v~nebi,
tak i~na zemi. Chléb náš vezdejší dej nám dnes. A~odpusť nám naše
viny, jako i~my odpouštíme našim viníkům. A~neuveď nás v pokušení, ale zbav
nás od zlého.}}

\newcommand{\trPater}{\translatioCantus{Otče, \grestar{} nemůže-li mě tento kalich minout, aniž bych jej pil, ať se stane tvá vůle!}}

\newcommand{\trAttende}{\translatioCantus{Ó, slyš a~shlédni k nám, smiluj se Pane, neboť jsme zhřešili tobě.}}
