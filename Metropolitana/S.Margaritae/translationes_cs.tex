%%%% Preklady jednotlivych zpevu (nektere se opakuji, a je dobre mit je
% vsechny na jedne hromade)

\newcommand{\trAntI}{\translatioCantus{Markéta, vznešená mučednice, je na příkaz prefektův nečinně vytažena ze žaláře a~přivedena před něj.}}

\newcommand{\trAntII}{\translatioCantus{Když žádné~\grestar{} lichocení soudcem nepohnulo, velice rozhněván ji poručil pálit loučemi.}}

\newcommand{\trAntIII}{\translatioCantus{Sluhové~\grestar{} hned začali pálit tělo útlé dívčiny, ale ta se modlila a~oheň se jí nedotknul.}}

\newcommand{\trAntIV}{\translatioCantus{Správce poručil přinést ohromnou nádobu s~vodou a~ponořit do ní spoutanou dívku.}}

\newcommand{\trAntV}{\translatioCantus{Chvála~\grestar{} Pánu a~jeho moci! Uvolnil pouta na rukou a~svou služebnici vysvobodil ze smrti.}}

\newcommand{\trCapituli}{\translatioCantus{Bratři: Ten, kdo se chlubí, ať se
chlubí v~Pánu.~\gredagger{} Není osvědčeným mužem ten, kdo doporučuje sám
sebe;~\grestar{} jím je ten, koho doporučuje Pán.}}

\newcommand{\trResp}{\translatioCantus{
Pomůže jí~\grestar{} Bůh svou přítomností.
\Vbardot{} Bůh je uprostřed ní, nepohne se.
\Vbardot{} Sláva Otci i~Synu i~Duchu Svatému.}}

\newcommand{\trVersiculi}{\translatioCantus{\Vbardot{} Milost se prostírá po tvých rtech.
\Rbardot{} Proto ti Bůh požehnal na věky.}}

\newcommand{\trAntMagnificat}{\translatioCantus{Velebme Pána,~\grestar{}
Spásu všech, v~duchu plesajíce a~tento svátek slavíce blažené Markéty,
jeho pokorné děvečky; abychom i~my na její prosbu smyli lejno hříchu.}}

\newcommand{\trDeprecatio}{\translatioCantus{Říkejme všichni z~celé duše
a~z~celé mysli: Vyslyš nás, Pane.
\Rbardot{} Vyslyš nás, Pane.
\Vbardot{} Pastýři našich duší, jenž neznáš spánku, bdi nad svým lidem.
\Vbardot{} Posvěť svůj lid svou neviditelnou přítomností a~ochraň jej před hrůzou noci.
\Vbardot{} Posilni, co je slabé, naprav, co je zlomené, oživ, co je mdlé, podpírej nás svou věrností, buduj svou láskou, očišťuj svou čistotou, osvěcuj svou moudrostí, uchovávej svým slitováním.
\Vbardot{} Dej, ať tvoji věrní očekávají Krista s~bdělostí; ať vytrvají v~lásce, ať žijí v~čistotě a~kázni.
\Vbardot{} A~v~štědrosti svého slitování nevylučuj ze svých zaslíbení žádného člověka.}}

\newcommand{\trOratio}{\translatioCantus{Bože, tvá síla se projevuje zvláště tehdy,
když působí v~tom, kdo je slabý:
tys posiloval svatou Markétu,
aby statečně snášela všechno mučení;
pomáhej i~nám, abychom na její přímluvu
vítězně prošli všemi zkouškami
do slávy tvého království.
Skrze našeho Pána…}}
