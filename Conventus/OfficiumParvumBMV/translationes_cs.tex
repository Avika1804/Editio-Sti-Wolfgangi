\newcommand{\trHymAveMarisStella}{
\begin{multicols}{6}
  \footnotesize
  \setlength{\parindent}{0pt}
  \setlength{\columnseprule}{0pt}
Zdrávas, hvězdo mořská,\\
Matko Boží štědrá,\\
ustavičná Panno,\\
šťastná nebes bráno.\columnbreak

Přijímajíc ``Ave''\\
ze rtů Gabriele,\\
vrať nám pokoj pravý,\\
měníc úděl Evy.\columnbreak

Rozvaž pouta vinným,\\ 
slepé vyveď ze tmy,\\
odvrať od nás vše zlé,\\
vypros všechno dobré.\columnbreak

Vypros, Matko naše,\\
milost od Ježíše,\\
neboť on nám z nebe\\
byl dán skrze tebe.\columnbreak

Panno nejvýš svatá,\\
mírná, vlídná, tichá,\\
veď nás cestou ctností,\\
ať jsme tiší, čistí.\columnbreak

Uč nás, Panno čistá,\\
následovat Krista,\\
u něho ať s tebou\\
najdem radost věčnou.
\end{multicols}
}

    slavné královno světa, 
    přimlouvej se za nás u Pána.}}

\newcommand{\trOratioVesperae}{\translatioCantus{
    Dopřej nám, prosíme, Pane Bože, těšit se ze stálého zdraví mysli i těla 
    a na přímluvu blažené Panny Marie nám dej vysvobození ze současných strastí 
    a účast na věčné radosti.}}

% COMPLETORIUM

\newcommand{\trHymMementoRerumConditor}{
Pamatuj, světa Původce,\\
žes kdysi naše tělo měl\\
a zrozen z Panny jeskyně\\ 
podobou ses přioděl.\\

Matko milosti, Maria,\\
sladká rodičko útěchy\\
před nepřítelem zachraň nás\\
a přijmi k sobě v smrti čas.\\

Ježíši, sláva tobě buď,\\
jenž z Panny jsi se narodil,\\
s Tebou i Otec, blahý Duch\\
navěky týž měj slávy díl.
}
