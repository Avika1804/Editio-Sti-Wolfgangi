\newcommand{\trAntReginaCaeli}{\translatioCantus{
    Vesel se nebes královno, aleluja, 
    neboť ten, kohos mohla nosit, aleluja, 
    jak řekl, vstal z mrtvých, aleluja. 
    Pros za nás Boha, aleluja.}}

    Jeho levici mám pod hlavou, jeho pravice mě objímá.}}

\newcommand{\trHymAveMarisStella}{
\begin{multicols}{6}
  \footnotesize
  \setlength{\parindent}{0pt}
  \setlength{\columnseprule}{0pt}
Zdrávas, hvězdo mořská,\\
Matko Boží štědrá,\\
ustavičná Panno,\\
šťastná nebes bráno.\columnbreak

Přijímajíc ``Ave''\\
ze rtů Gabriele,\\
vrať nám pokoj pravý,\\
měníc úděl Evy.\columnbreak

Rozvaž pouta vinným,\\ 
slepé vyveď ze tmy,\\
odvrať od nás vše zlé,\\
vypros všechno dobré.\columnbreak

Vypros, Matko naše,\\
milost od Ježíše,\\
neboť on nám z nebe\\
byl dán skrze tebe.\columnbreak

Panno nejvýš svatá,\\
mírná, vlídná, tichá,\\
veď nás cestou ctností,\\
ať jsme tiší, čistí.\columnbreak

Uč nás, Panno čistá,\\
následovat Krista,\\
u něho ať s tebou\\
najdem radost věčnou.
\end{multicols}
}

\newcommand{\trAntMagnificat}{\translatioCantus{
    Blažená Matko a neporušená Panno, 
    slavné královno světa, 
    přimlouvej se za nás u Pána.}}

\newcommand{\trOratioVesperae}{\translatioCantus{
    Dopřej nám, prosíme, Pane Bože, těšit se ze stálého zdraví mysli i těla 
    a na přímluvu blažené Panny Marie nám dej vysvobození ze současných strastí 
    a účast na věčné radosti.}}

% COMPLETORIUM

\newcommand{\trHymMementoRerumConditor}{
Pamatuj, světa Původce,\\
žes kdysi naše tělo měl\\
a zrozen z Panny jeskyně\\ 
podobou ses přioděl.\\

Matko milosti, Maria,\\
sladká rodičko útěchy\\
před nepřítelem zachraň nás\\
a přijmi k sobě v smrti čas.\\

Ježíši, sláva tobě buď,\\
jenž z Panny jsi se narodil,\\
s Tebou i Otec, blahý Duch\\
navěky týž měj slávy díl.
}

\newcommand{\trCapitEgoMater}{\translatioCantus{
    Já jsem matka líbezného milování, † bázně i poznání * i svaté naděje.}}
\newcommand{\trVersOraProNobis}{\translatioCantus{
    ℣. Pros za nás svatá Boží Rodičko.
    ℟. Abychom se stali hodni zaslíbení Kristova.}}

\newcommand{\trCapitEcceVirgo}{\translatioCantus{
    Hle Panna počne a porodí syna † a nazve jeho jméno Emmanuel. * 
    Bude jíst máslo a med, aby uměl odvrhnout zlé a vybrat si dobré.}}
\newcommand{\trVersOraProNobis}{\translatioCantus{
    ℣. Anděl Páně zvěstoval Marii
    ℟. a počala z Ducha Svatého.}}

\newcommand{\trAntSubTuum}{\translatioCantus{
    K tvé ochraně se utíkáme, svatá Boží Rodičko, 
    nepomíjej naše prosby za běžné potřeby, 
    ale vždy nás zachraň z každého nebezpečí, 
    Slavná a požehnaná Panno.}}

\newcommand{\trOratioCompletorium}{\translatioCantus{
    Slavná přímluva slavné a blahoslavené navždy Panny Marie 
    nechť nás zachrání pro život věčný.}}
