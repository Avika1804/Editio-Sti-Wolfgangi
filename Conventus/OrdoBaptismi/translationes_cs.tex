\newcommand{\trIntroitus}{\translatioCantus{Vy všichni, kdo žízníte,~\grestar{}
přijďte k vodě, praví Hospodin, i když nemáte peníze, přijďte, pijte s radostí.
\textit{\color{red}Žl.} Lide můj, poslyš můj zákon; slovům mých úst nastav
uši.}}

\newcommand{\trAlleluia}{\translatioCantus{Aleluja. \Vbardot{}
Pán přišel ze Sinaje do svatyně, vystoupil na výšinu, pochytal zajatce.}}

\newcommand{\trCommunio}{\translatioCantus{Vy, kdož jste v Krista byli
pokřtěni, Kriste jste oblékli, aleluja. \Vbardot{}
Hlas Hospodinův rachotí nad vodami, Hospodin, Bůh velebnosti, hřímá,
\grestar{} (hřímá) Hospodin nad spoustami vodstva.}}

\newcommand{\trDaPacem}{\translatioCantus{Dej pokoj, Pane, v našich dnech,
protože nikdo jiný za nás nebojuje, než Ty, Bože náš.}}
