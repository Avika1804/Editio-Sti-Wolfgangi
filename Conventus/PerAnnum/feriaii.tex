\pars{Preces.}

\sineinitiali{}{temporalia/tonusprecumnovum.gtex}

\ifx\preces\undefined
\ifx\lauda\undefined
\else
\noindent Christum magnificémus, \gredagger{} plenum grátia et Spíritu Sancto, \grestar{} et fidénter eum implorémus:

\Rbardot{} Spíritum tuum da nobis, Dómine.

\noindent Concéde nobis diem istum iucúndum, pacíficum et sine mácula, \grestar{} ut, ad vésperam perdúcti, cum gáudio et mundo corde te collaudáre valeámus.

\Rbardot{} Spíritum tuum da nobis, Dómine.

\noindent Sit hódie splendor tuus super nos, \grestar{} et opus mánuum nostrárum dírige.

\Rbardot{} Spíritum tuum da nobis, Dómine.

\noindent Osténde fáciem tuam super nos ad bonum in pace, \grestar{} ut hódie manu tua válida contegámur.

\Rbardot{} Spíritum tuum da nobis, Dómine.

\noindent Réspice propítius omnes, qui oratiónibus nostris confídunt, \grestar{} eos adímple bonis ánimæ et córporis univérsis.

\Rbardot{} Spíritum tuum da nobis, Dómine.
\fi
\ifx\laudb\undefined
\else
\noindent Salvátor noster fecit nos regnum et sacerdótium, \gredagger{} ut hóstias Deo acceptábiles offerámus. \grestar{} Grati ígitur eum invocémus:

\Rbardot{} Serva nos in tuo ministério, Dómine.

\noindent Christe, sacérdos ætérne, \gredagger{} qui sanctum pópulo tuo sacerdótium concessísti, \grestar{} concéde, ut spiritáles hóstias Deo acceptábiles iúgiter offerámus.

\Rbardot{} Serva nos in tuo ministério, Dómine.

\noindent Spíritus tui fructus nobis largíre propítius, \grestar{} patiéntiam, benignitátem et mansuetúdinem.

\Rbardot{} Serva nos in tuo ministério, Dómine.

\noindent Da nobis te amáre, \gredagger{} ut te, qui es cáritas, possideámus, \grestar{} et bene ágere, ut per vitam étiam nostram te laudémus.

\Rbardot{} Serva nos in tuo ministério, Dómine.

\noindent Quæ frátribus nostris sunt utília, \gredagger{} nos quǽrere concéde, \grestar{} ut salútem facílius consequántur.

\Rbardot{} Serva nos in tuo ministério, Dómine.
\fi
\ifx\laudc\undefined
\else
\noindent Deum, qui hómines in mundo pósuit, \gredagger{} ut concórditer ad suam glóriam opus fíeret ab illis, \grestar{} eníxe rogémus:

\Rbardot{} Fac nos te glorificáre, Dómine.

\noindent Creátor universórum Deus, \gredagger{} benedícimus tibi, \grestar{} qui nobis mundi bona dedísti vitámque nostram hucúsque servásti.

\Rbardot{} Fac nos te glorificáre, Dómine.

\noindent Réspice nos, \gredagger{} qui labórem cotidiánum suscípimus, \grestar{} et óperis tui partícipes nos tuæ voluntáti confórma.

\Rbardot{} Fac nos te glorificáre, Dómine.

\noindent Opus nostrum hodiérnum sic redde nostris frátribus fructuósum, \grestar{} ut cum eis et pro eis terrénam civitátem, quæ tibi pláceat, ædificémus.

\Rbardot{} Fac nos te glorificáre, Dómine.

\noindent Nobis et ómnibus, quibus hódie occurrémus, adésto, \grestar{} gáudium largíre et pacem.

\Rbardot{} Fac nos te glorificáre, Dómine.
\fi
\ifx\laudd\undefined
\else
\noindent Christum, qui exáudit et salvos facit sperántes in se, \grestar{} precémur acclamántes:

\Rbardot{} Te laudámus, in te sperámus, Dómine.

\noindent Grátias ágimus tibi, \gredagger{} qui dives es in misericórdia, \grestar{} propter nímiam caritátem, qua dilexísti nos.

\Rbardot{} Te laudámus, in te sperámus, Dómine.

\noindent Qui omni témpore in mundo cum Patre operáris, \grestar{} nova fac ómnia per Spíritus Sancti virtútem.

\Rbardot{} Te laudámus, in te sperámus, Dómine.

\noindent Aperi óculos nostros et fratrum nostrórum, \grestar{} ut videámus hódie mirabília tua.

\Rbardot{} Te laudámus, in te sperámus, Dómine.

\noindent Qui nos hódie ad tuum servítium vocas, \grestar{} nos erga fratres multifórmis grátiæ tuæ fac minístros.

\Rbardot{} Te laudámus, in te sperámus, Dómine.
\fi
